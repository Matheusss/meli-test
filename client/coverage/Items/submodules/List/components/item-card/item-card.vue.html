<template>
  <div class="item-list-card">
    <img
      :src="item.picture"
      :alt="item.title"
      height="180"
      width="180"
      @click="navigateToDetails">
    <div class="item-infos">
      <p class="item-price" @click="navigateToDetails">
        {{ item.price.decimals | currency(item.price.currency) }}
      </p>
      <p>{{ item.title }}</p>
      <p>{{ item.condition }}</p>
    </div>
  </div>
</template>
 
<script>
 
export default {
  name: 'ItemListCard',
  props: {
    item: {
      type: Object,
      required: true,
    },
  },
  methods: {
    navigateToDetails() {
      this.$router.push({
        name: 'items.details',
        params: {
          id: this.item.id,
        }
      })
    }
  },
}
</script>
